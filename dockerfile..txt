FROM php:8.2-apache

# Install system dependencies
RUN apt-get update && apt-get install -y \
    git \
    curl \
    libpng-dev \
    libonig-dev \
    libxml2-dev \
    zip \
    unzip

# Clear cache
RUN apt-get clean && rm -rf /var/lib/apt/lists/*

# Install PHP extensions
RUN docker-php-ext-install mysqli pdo pdo_mysql mbstring exif pcntl bcmath gd

# Enable Apache mod_rewrite (Header module is not available in this image)
RUN a2enmod rewrite

# Copy application files
COPY . /var/www/html/

# Set proper permissions
RUN chown -R www-data:www-data /var/www/html/
RUN chmod 755 /var/www/html/
RUN chmod 666 /var/www/html/users.json 2>/dev/null || true

# Create error log file with proper permissions
RUN touch /var/www/html/error.log && chmod 666 /var/www/html/error.log

# Create health check file
RUN echo "<?php echo 'OK'; ?>" > /var/www/html/health.php

# Expose port
EXPOSE 80

CMD ["apache2-foreground"]
